streamlit>=1.35

# ── Bethe-Bloch engine — Python driver (Engine 2) ────────────────────────────
# Pure Python/NumPy CSDA transport — no additional packages needed.
# MS=OFF: analytical exact CSDA, sub-second for 600k muons.
# MS=ON:  vectorized NumPy stepping with Highland MCS.
# Driver file: gui/ucmuon_bb_driver.py

# ── UCMuon Stochastic engine (Engine 4) ──────────────────────────────────────
# Self-developed pure-Python stochastic transport — no additional packages needed.
# Uses only numpy and scipy (already listed above).
# Driver files: gui/ucmuon_stochastic_driver.py, gui/gui_stochastic_engine.py

# ── Backward MC Flux Integrator (Engine 5) ───────────────────────────────────
# No additional packages needed beyond numpy and scipy.
# Driver file: gui/ucmuon_backward_mc.py

# ── PUMAS backward/forward MC (Engine 7) ─────────────────────────────────────
# C binary — no Python packages needed beyond numpy (already above).
# Requires: pumas-master/ directory in project root (LGPL, Niess et al. 2017).
# Build:     make pumas        (runs automatically if pumas-master/ is present)
# First run: builds a physics dump in bin/pumas_StandardRock.pumas (~10 s)
# Subsequent runs load the dump and start instantly.
# NOT required for engines 1–5 — entirely optional.

# ── UCMuon Terrain / DEM ray-tracing (Engine 6 — optional) ──────────────────
# Required for DEM loading and coordinate reprojection.
# Install: pip install rasterio
#
# rasterio>=1.3

# ── PROPOSAL muon transport (Engine 3 — optional) ────────────────────────────
# Full stochastic MC (Koehne et al. 2013, Alameddine et al. 2024).
#
# IMPORTANT: incompatible with Anaconda/miniforge (pybind11 ABI mismatch).
# Use a system Python venv:
#
#   /usr/bin/python3 -m venv ~/venvs/ucmuon
#   source ~/venvs/ucmuon/bin/activate
#   pip install -r requirements.txt
#   pip install proposal
#
# proposal>=7.6.2
